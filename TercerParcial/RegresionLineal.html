#  Importar libreras
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
from sklearn.linear_model import LinearRegression

#  Dataset base simulado
data = pd.DataFrame({
    'Word count': [500, 1000, 1500, 2000, 2500, 3000, 3500],
    '# Shares': [5000, 9000, 15000, 20000, 27000, 32000, 35000]
})

#  Preparar datos de entrenamiento
X_train = data[['Word count']].values
y_train = data['# Shares'].values

#  Crear y entrenar modelo
modelo = LinearRegression()
modelo.fit(X_train, y_train)

#  Pedir al usuario que ingrese el punto 
x_usuario = float(input(" Ingresa el valor de  (cantidad de palabras del artculo): "))

#  Generar la prediccin Y
y_predicho = modelo.predict([[x_usuario]])
print(f" Prediccin: Con {x_usuario} palabras, se estima que el artculo ser compartido {int(y_predicho[0])} veces.")



print(f" El punto ({x_usuario}, {int(y_predicho[0])}) est en el: {cuadrante}")

#  Dibujar grfico con la prediccin
# Lnea de regresin
x_linea = np.linspace(400, 3600, 100).reshape(-1, 1)
y_linea = modelo.predict(x_linea)

# Mostrar grfico
plt.scatter(X_train, y_train, color='blue', label='Datos reales')
plt.plot(x_linea, y_linea, color='red', linewidth=2, label='Regresin Lineal')
plt.scatter(x_usuario, y_predicho, color='green', s=120, label='Punto ingresado')
plt.xlabel('Cantidad de Palabras')
plt.ylabel('Compartido en Redes')
plt.title('Regresin Lineal + Prediccin por Teclado')
plt.legend()
plt.grid(True)
plt.show()
